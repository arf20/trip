77void
78logging_log(loglevel_t level, const char *component, const char *fmt, ...)
79{
80 char logbuff[4096];
81 va_list args;
82
83 if (level > g_loglevel)
84 return;
85
86 va_start(args, fmt);
87 vsnprintf(logbuff, 4096, fmt, args);
88 va_end(args);
89
90 fprintf(g_logf, "[%s %s %s] %s\n", timestr(), loglevel_strs[level],
91 component, logbuff);
92}
